async function transmitNote() { const val = document.getElementById('noteInput').value; if(!val) return; const status = document.getElementById('status-text'); const btn = document.getElementById('sendBtn'); status.innerText = "TRANSMITTING..."; status.className = "text-yellow-500 font-bold text-[9px] animate-pulse uppercase tracking-widest italic"; btn.disabled = true; try { // Forzamos la ruta completa para evitar el Connection Error const response = await fetch('https://montcode-dr.vercel.app/api/pusher-relay', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }, body: JSON.stringify({ message: val }) }); if (response.ok) { status.innerText = "SENT SUCCESSFUL ✅"; status.className = "text-[#ccff00] font-bold text-[9px] uppercase tracking-widest italic"; document.getElementById('noteInput').value = ""; } else { const errorText = await response.text(); console.error("Detalle del error:", errorText); status.innerText = "SERVER REJECTED ❌"; } } catch (e) { console.error("Error de red:", e); status.innerText = "CONNECTION ERROR ❌"; } finally { btn.disabled = false; } }
